1146  inline void dg2eijk(unsigned int dg_index, unsigned int &e, unsigned int &i, unsigned int &j, unsigned int &k) const
1147  {
1148  e = dg_index / m_uiNpE;
1149  k = 0;
1150  j = 0;
1151  i = 0;
1152 
1153  //std::cout<<"e: "<<e<<std::endl;
1154  if (dg_index > e * m_uiNpE)
